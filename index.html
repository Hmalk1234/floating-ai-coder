-- Roblox Game Script Example: Pet System controller
local Players = game:GetService("Players")
local DataStoreService = game:GetService("DataStoreService")
local PetDataStore = DataStoreService:GetDataStore("PetSaveSystem")

local function onPlayerAdded(player)
    local leaderstats = Instance.new("Folder")
    leaderstats.Name = "leaderstats"
    leaderstats.Parent = player

    local coins = Instance.new("IntValue")
    coins.Name = "Coins"
    coins.Value = 100
    coins.Parent = leaderstats

    local equippedPet = Instance.new("StringValue")
    equippedPet.Name = "EquippedPet"
    equippedPet.Value = "None"
    equippedPet.Parent = player
end

Players.PlayerAdded:Connect(onPlayerAdded)
